\frametitle{انتشار چندمسیره و فیدینگ}

\textbf{توضیح:} سیگنال از مسیرهای مختلفی به گیرنده می‌رسد.

\vspace{0.3cm}
\textbf{انواع محوشدگی (\LR{Fading}):}
\begin{itemize}
    \item \textbf{بزرگ‌مقیاس (\LR{Large-scale}):} ناشی از موانع بزرگ و سایه (\LR{Shadowing}).
    \item \textbf{کوچک‌مقیاس (\LR{Small-scale}):} نوسانات سریع ناشی از تداخل امواج.
\end{itemize}

\begin{myblock}{راهکار مقابله}
استفاده از \LR{MIMO} و \LR{OFDM} در شبکه‌های \LR{4G/5G}.
\end{myblock}
\clearpage

% --- اسلاید ۶: بودجه لینک ---
\frametitle{بودجه لینک (\LR{Link Budget})}

\textbf{هدف:} محاسبه تمام توان‌ها و تلفات برای تضمین اتصال.

\begin{equation}
    P_{RX} = P_{TX} + G_{TX} + G_{RX} - L_{Path} - L_{Misc}
\end{equation}

\vspace{0.3cm}
\begin{itemize}
    \item $P_{RX}$: توان دریافتی (باید بیشتر از حساسیت گیرنده باشد).
    \item $L$: مجموع تمام تلفات (مسیر، کابل، کانکتور).
\end{itemize}
